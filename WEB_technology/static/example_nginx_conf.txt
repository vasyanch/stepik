# расположение этого файла: /etc/nginx/nginx.conf  при условии что nginx установлен на машину

user www www;
error_log /var/log/nginx.error_log info;
http {

  include        conf/mime.types;
  default_type   application/octet-stream;
  log_format     simple '$remote_addr $request $status';
  server {                                                   # начало настроек virtual host

    listen         one.example.com;                          # указывается IP-адрес и порт для TCP-соединения 
    server_name    one.example.com www.one.example.com;
    access_log     /var/log/nginx.access_log simple;
    location / {                                             # начало настроек группы url-ов
    
      root           /www/one.example.com;
    }
    location ~* ^.+\.(jpg|jpeg|gif)$ {
    
      root           /www/images;
      access_log     off;
      expires        30d'
    }
  }
}
